### Scroll Behavior Implementation

**Transparent to Solid Transition:**
Implement scroll-based header transformation using the following technical approach:

1. **Initial State (Top of Page):**
   - `position: fixed`
   - `background-color: transparent`
   - `padding: 25px 50px`
   - No box-shadow

2. **Scrolled State (After threshold):**
   - `background-color: rgba(255, 255, 255, 0.95)`
   - `backdrop-filter: blur(10px)`
   - `padding: 15px 50px`
   - `box-shadow: 0 2px 15px rgba(0, 0, 0, 0.1)`

3. **JavaScript Implementation:**
   - Use `requestAnimationFrame` for performance optimization
   - Add `{ passive: true }` to scroll event listener
   - Set scroll threshold at 50px
   - Toggle `.scrolled` class based on `window.scrollY`

```javascript
const header = document.querySelector('.main-header');
let ticking = false;

function updateHeader() {
  const scrollPosition = window.scrollY;
  const threshold = 50;
  
  if (scrollPosition > threshold) {
    header.classList.add('scrolled');
  } else {
    header.classList.remove('scrolled');
  }
  ticking = false;
}

window.addEventListener('scroll', function() {
  if (!ticking) {
    window.requestAnimationFrame(updateHeader);
    ticking = true;
  }
}, { passive: true });

Hide on Scroll Down / Show on Scroll Up
Implement directional scroll detection:

let lastScrollPosition = 0;

function updateHeader() {
  const currentScrollPosition = window.scrollY;
  
  // Add scrolled class after threshold
  if (currentScrollPosition > 50) {
    header.classList.add('scrolled');
  } else {
    header.classList.remove('scrolled');
  }
  
  // Hide/show based on scroll direction
  if (currentScrollPosition > lastScrollPosition && currentScrollPosition > 100) {
    // Scrolling DOWN - hide header
    header.classList.add('header-hidden');
  } else {
    // Scrolling UP - show header
    header.classList.remove('header-hidden');
  }
  
  lastScrollPosition = currentScrollPosition;
  ticking = false;
}
